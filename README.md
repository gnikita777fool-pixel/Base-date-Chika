# Основные знания по базам данных:

## Создание таблицы:

**CREATE TABLE <название таблиц> - создает таблицу**

CREATE TABLE <название таблиц> ( - /*создание таблицы с переменными*/
<название переменной> INT - /*создает переменную*/
<название переменной> VARCHAR(265) - /*создает строчную переменную*/

Пример кода:

CREATE TABLE student (
pr_name VARCHAR(265), 
age INT,
tax INT);

## Внесение данных в таблицу:

INSERT INTO <название таблицы> (название переменной, ...,) Values

INSERT INTO student (pr_name, age, tax) Values 
('Maria',18,110), 
('Sveta',19,100), 
('Olya',20,111);

select - позволяет выбирать таблицы и данные из таблиц
  * from - пишеться вместе с select, позволяет выбирать конкретные таблицы
  where - пишеться вместе с select, позволяет выводить данные если оно совпадает с условием

Пример кода:
select <таблица, ... ,> * from <столбец>

grow by - позволяет группировать данные из таблиц
heavy by - позволяет фильтровать данные из группировки
order by - позволяет сортировать данные по стоблцу 
ask - сортировка по возрастанию 
desk - сортировка данных по убыванию
limit - количестко значений которых надо вывести

"right or left" join - соединение таблиц 
right - сторона с которой начинается вывод
left - сторона с которой начинается вывод

AS - условное изменение названия таблицы
